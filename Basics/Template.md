## Here is a template that can make it easier to create an effect:
 It shows the example of a program structure that you can complete with your code. 
 
 
 ``` Code
// @Maintainer ...          // optional
// @Released ...
// @Author ...
// @Created ...
// @see ...                 // image link recommended
// @see ...                 // video link optional
//-----------------------------------------------------------------------------------------//
// Lightworks user effect ...filename.fx
// Effect Description ...
// ......................
// ......................
//-----------------------------------------------------------------------------------------//

int _LwksEffectInfo
<
   string EffectGroup = "GenericPixelShader";
   string Description = "...";
   string Category    = "...";
   string SubCategory = "...";     // optional
   string Notes = "...";           // optional
> = 0;

//--------------------------------------------------------------//
// Inputs
//--------------------------------------------------------------//

...

//--------------------------------------------------------------//
// Samplers
//--------------------------------------------------------------//

...

//--------------------------------------------------------------//
// Parameters
//--------------------------------------------------------------//

...

//--------------------------------------------------------------//
// Definitions and declarations
//--------------------------------------------------------------//

...

//--------------------------------------------------------------//
// Functions
//--------------------------------------------------------------//

float2 fn_some_func (...)
{

...

}

//--------------------------------------------------------------//
// Shaders
//--------------------------------------------------------------//

float4 ps_main (...) : COLOR
{

...

}

//--------------------------------------------------------------//
// Techniques
//--------------------------------------------------------------//

technique MyEffect
{
   pass P_1
   {
      PixelShader = compile PROFILE ps_main ();
   }
}
```
